15// Base template struct for finding the best common type among multiple types
16template <typename T, typename... Ts>
17struct InnerFindCommonType;
18
19// Specialization of InnerFindCommonType for a single type
20template <typename T>
21struct InnerFindCommonType<T>
22{
23 using type = T; // Common type is the single type itself
24};
25
26// Specialization of InnerFindCommonType for multiple types
27template <typename T, typename U, typename... Ts>
28struct InnerFindCommonType<T, U, Ts...>
29{
30 // Determines the common type among T, U, and Ts. If U and all Ts are convertible to T, the common type is T.
31 // Otherwise, it recursively determines the common type among U and Ts.
32 using type = std::conditional_t<(std::convertible_to<U, T> && ... && std::convertible_to<Ts, T>),
33 typename InnerFindCommonType<T, Ts...>::type,
34 typename InnerFindCommonType<U, Ts...>::type>;
35};
