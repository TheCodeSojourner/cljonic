58template <typename F, typename... Fs>

61// Specialization of AreInvocableWithReturn for multiple functions
62template <typename F1, typename F2, typename... Fs>
63struct AreInvocableWithReturn<F1, F2, Fs...>
64{
65 // Value is true if F1 is invocable with the return type of F2 and the rest of the functions follow the same rule
