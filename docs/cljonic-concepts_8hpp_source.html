12// Base template struct for finding the best common type among multiple types
13template <typename T, typename... Ts>
14struct InnerFindCommonType;
15
16// Specialization of InnerFindCommonType for a single type
17template <typename T>
18struct InnerFindCommonType<T>
19{
20 using type = T; // Common type is the single type itself
21};
22
23// Specialization of InnerFindCommonType for multiple types
24template <typename T, typename U, typename... Ts>
25struct InnerFindCommonType<T, U, Ts...>
26{
27 // Determines the common type among T, U, and Ts. If U and all Ts are convertible to T, the common type is T.
28 // Otherwise, it recursively determines the common type among U and Ts.
29 using type = std::conditional_t<(std::convertible_to<U, T> && ... && std::convertible_to<Ts, T>),
30 typename InnerFindCommonType<T, Ts...>::type,
31 typename InnerFindCommonType<U, Ts...>::type>;
32};
